Oracle 12c单实例安装手册

本文最后更新于：1 小时前

12c单实例安装手册

OS Version：rhel6.9
内存：4G
硬盘：100G
分区： /boot 200M; swap 4096M; /u01 50G; / 40G
处理器：1cpu，2核心
网卡：hostonly / NAT

1 获取介质

下载链接

2 操作系统检查及调整

2.1 配置本地主机名解析

[root@yzy ~]# cat /etc/sysconfig/network
NETWORKING=yes
HOSTNAME=yzy
[root@yzy ~]#
[root@yzy ~]# cat >>/etc/hosts<< EOF
> 192.168.0.111 yzy
> EOF

2.2 修改内核参数

修改一些内核参数，如果已经存在，它会取最高值

内存 = 4G

Controls the maximum shared segment size, in bytes

kernel.shmmax = 410241024*1024-1 = 41947234303

# redhat 6.9 写在/etc/sysctl.d/97-oracledatabase-sysctl.conf
# redhat 6.4 写在/etc/sysctl.conf

cat>>/etc/sysctl.d/97-oracledatabase-sysctl.conf<<EOF
fs.suid_dumpable=1
fs.aio-max-nr = 1048576
fs.file-max = 6815744
kernel.shmall = 2097152
kernel.shmmax = 41947234303
kernel.shmmni = 4096
kernel.sem = 250 32000 100 128
net.ipv4.ip_local_port_range = 9000 65500
net.core.rmem_default = 262144
net.core.rmem_max = 4194304
net.core.wmem_default = 262144
net.core.wmem_max = 1048586
EOF

sysctl -p /etc/sysctl.d/97-oracledatabase-sysctl.conf    # 使生效

设置oracle用户shell限制

cat>>/etc/security/limits.conf<<EOF
oracle soft nproc 2047
oracle hard nproc 16384
oracle soft nofile 1024
oracle hard nofile 65536
oracle soft stack 10240
oracle hard stack 32768
EOF

cat>>/etc/pam.d/login<<EOF
session    required     pam_limits.so
EOF

2.3 设置/dev/shm

添加红字部分，设置/dev/shm的size

[root@yzy ~]# vim /etc/fstab
tmpfs                   /dev/shm                tmpfs   defaults,size=2G        0 0
devpts                  /dev/pts                devpts  gid=5,mode=620  0 0
sysfs                   /sys                    sysfs   defaults        0 0
proc                    /proc                   proc    defaults        0 0
/dev/sdb2               /u01                   ext4    defaults        0 0

[root@yzy ~]# mount -o remount /dev/shm
[root@yzy ~]# df -h /dev/shm/
Filesystem      Size  Used Avail Use% Mounted on
tmpfs           2.0G   72K  2.0G   1% /dev/shm

2.4 创建oracle用户和组

groupadd oinstall
groupadd dba
groupadd oper

useradd  -g oinstall -G dba,oper oracle

echo "oracle"|passwd --stdin oracle

2.5 创建安装目录

mkdir -p  /u01/app/oracle
chown -R oracle:oinstall  /u01
chmod -R 775  /u01/app/oracle

2.6 配置oracle用户环境变量

su - oracle
cat >>/home/oracle/.bash_profile  <<EOF
export ORACLE_BASE=/u01/app/oracle
export ORACLE_HOME=/u01/app/oracle/product/12.2.0/dbhome_1
export PATH=\$ORACLE_HOME/bin:\$PATH
EOF

source ~/.bash_profile

2.8 yum安装所需的包

yum -y install libst*
yum -y install gcc*
yum -y install elfu*
yum -y install libaio*
yum -y install compat*
yum -y install ksh

2.9 关闭NetworkManager和防火墙

chkconfig NetworkManager off
chkconfig iptables off

# 检查状态
chkconfig NetworkManager --list
chkconfig iptables --list

3 安装软件

3.1 解压介质

su - oracle
unzip linuxx64_12201_database.zip

3.2 安装

宿主机打开 Xmanager - Passive 服务

export DISPLAY=192.168.0.111:0.0     # 宿主机ip地址

xhost +

进入软件目录

./runInstaller

3.3 netca配置监听

netca

3.4 dbca建库

dbca

4 其他配置

4.1 配置ORACLE_SID变量

[oracle@yzy ~]$ cat>>/home/oracle/.bash_profile<<EOF
> export ORACLE_SID=zydb
> export NLS_LANG=AMERICAN_AMERICA.AL32UTF8
> EOF
[oracle@yzy ~]$ source .bash_profile
[oracle@yzy ~]$ echo $ORACLE_SID
zydb

4.2 sqlplus配置

cat>>$ORACLE_HOME/sqlplus/admin/glogin.sql<<EOF
set linesize 120
set pagesize 999
define _editor=vi
set sqlprompt "_user'@'_connect_identifier> "
EOF

4.3 rlwrap配置

yum -y install readline*

# 上传rlwrap包
tar -xvf rlwrap-0.37.tar.gz
cd rlwrap-0.37
./configure

make && make install
which rlwrap

cat>>/home/oracle/.bash_profile<<EOF
alias sqlplus='/usr/local/bin/rlwrap sqlplus'
alias rman='/usr/local/bin/rlwrap rman'
EOF

# oracle用户
source ~/.bash_profile
