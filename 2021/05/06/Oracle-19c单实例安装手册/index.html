Oracle 19c单实例安装手册

本文最后更新于：1 天前

19c单实例安装手册

OS Version：CentOS 7.6
内存：8G
硬盘：100G
分区： /boot 200M; swap 4096M; /u01 50G; / 40G
处理器：1cpu，2核心
网卡：hostonly / NAT

1 获取介质

下载链接

2 操作系统检查及调整

硬件、操作系统、服务器配置、用户环境配置及存储配置根据官方文档要求进行配置

2.1 安装所需的包

从18c 开始，可通过手动安装和自动配置两种方式解决操作系统依赖安装

2.1.1 手动安装

可通过网络yum源进行安装 (CentOS)

也可通过配置本地yum源仓库进行安装

yum install libst* -y
yum install gcc* -y
yum install libaio* -y
yum install glibc* -y
yum install compat*
yum install libx* -y
yum install libX* y
yum install make* -y
yum install sysstat*

2.1.2 自动配置

preinstall包除了自动安装所需的包之外，还会创建相应的group和user

# 获取rpm包
curl -o oracle-database-preinstall-19c-1.0-1.el7.x86_64.rpm https://yum.oracle.com/repo/OracleLinux/OL7/latest/x86_64/getPackage/oracle-database-preinstall-19c-1.0-1.el7.x86_64.rpm

# 安装rpm包
yum -y localinstall oracle-database-preinstall-19c-1.0-1.el7.x86_64.rpm

2.2 检查内核版本及参数调整

2.2.1 内核版本检查

cat /etc/oracle-release   # oracle linux os
cat /etc/redhat-release   # redhat os
cat /etc/os-release
lsb_release -id
uname -r

2.2.2 内核参数配置

/etc/sysctl.conf

cat>>/etc/sysctl.conf<<EOF
net.ipv4.ip_local_port_range = 9000 65500
net.ipv4.tcp_rmem = 4096 87380 4194304
net.ipv4.tcp_wmem = 4096 16384 4194304
fs.aio-max-nr = 1048576   # 限制并发未完成的请求，文件系统最大异步IO，设置避免io子系统故障
fs.file-max = 6815744   # 决定了系统中所允许的文件句柄最大数目，文件句柄设置代表linux系统中可以打开的文件的数量
kernel.shmall = 2097152
kernel.shmmax = 4294967295
kernel.shmmni = 4096    # 表示单个共享内存段的最小值
kernel.sem = 250 32000 100 128    # 该参数表示可以使用的信号量
net.ipv4.ip_local_port_range = 9000 65500
net.core.rmem_default = 262144   # 表示套接字接收缓冲区大小的缺省值
net.core.rmem_max = 4194304      # 表示套接字接收缓冲区大小的最大值
net.core.wmem_default = 262144   # 表示套接字发送缓冲区大小的缺省值
net.core.wmem_max = 1048576      # 表示套接字发送缓冲区大小的最大值
EOF

sysctl -p

/etc/security/limits.conf

cat>>/etc/security/limits.conf<<EOF
oracle soft nofile 1024
oracle hard nofile 65536
oracle soft nproc 2047
oracle hard nproc 16384
oracle soft stack 10240
oracle hard stack 32768
EOF

关闭透明大页

透明大页会在运行时导致内存分配延迟

为了避免性能问题，Oracle建议在所有Oracle数据库服务器上禁用透明大页

# rhel
cat /sys/kernel/mm/redhat_transparent_hugepage/enabled 

# other
cat /sys/kernel/mm/transparent_hugepage/enabled
[always] madvise never    # 返回该结果表示当前为always模式，即开启状态

# 关闭 (for ol7 & rhel7)
vim /etc/default/grub
GRUB_TIMEOUT=5
GRUB_DISTRIBUTOR="$(sed 's, release .*$,,g' /etc/system-release)"
GRUB_DEFAULT=saved
GRUB_DISABLE_SUBMENU=true
GRUB_TERMINAL_OUTPUT="console"
GRUB_CMDLINE_LINUX="rhgb quiet numa=off transparent_hugepage=never "  # 添加绿色部分
GRUB_DISABLE_RECOVERY="true"

# 生成grub.cfg文件
grub2-mkconfig -o /boot/grub2/grub.cfg

2.2.3 关闭防火墙及selinux

关闭防火墙

# 查看防火墙状态
systemctl list-unit-files|grep firewalld

# 关闭及禁用
systemctl stop firewalld.service
systemctl status firewalld.service
systemctl disable firewalld.service

禁用selinux

/etc/selinux/config

vim /etc/selinux/config
SELINUX=disabled

# 重启后生效

# 查看状态
getenforce

3 配置组、用户、目录

如果在2.1步骤中，执行的是2.1.2 自动配置，则所需的组和用户已经自动创建，可忽略3.1及3.2

3.1 创建组

标红的组可不用创建，根据具体需求进行选择，但oinstall和dba组必须要有

groupadd -g 54321 oinstall
groupadd -g 54322 dba
groupadd -g 54323 oper
groupadd -g 54324 backupdba
groupadd -g 54325 dgdba
groupadd -g 54326 kmdba
groupadd -g 54330 racdba

3.2 创建用户

如果创建了标红的组，则需要将该组添加为属组

useradd -u 54321 -g oinstall -G dba,oper,backupdba,dgdba,kmdba,racdba,asmdba oracle

echo "oracle" | passwd --stdin oracle

3.3 创建目录

根据需求创建，路径不做要求

mkdir -p /u01/app
chown -R oracle:oinstall /u01/app
chmod -R 775 /u01/app

mkdir -p /u01/app/oraInventory
chown -R oracle:oinstall /u01/app/oraInventory
chmod -R 775 /u01/app/oraInventory

mkdir -p /u01/app/oracle/cfgtoollogs
mkdir -p /u01/app/oracle/product/19.3.0/dbhome_1
chown -R oracle:oinstall /u01/app/oracle
chmod -R 775 /u01/app/oracle

4 安装软件

4.1 配置oracle用户环境变量

su - oracle
vim .bash_profile
export ORACLE_BASE=/u01/app/oracle
export ORACLE_HOME=$ORACLE_BASE/product/19.3.0/dbhome_1
export ORACLE_SID=zydb
export PATH=$ORACLE_HOME/bin:$PATH
export LD_LIBRARY_PATH=$ORACLE_HOME/lib:$LD_LIBRARY_PATH

source .bash_profile

4.2 解压软件

在19c中，要求必须将软件解压至 ORACLE_HOME下

unzip LINUX.X64_193000_db_home\ .zip -d $ORACLE_HOME

4.3 安装软件

启动Xmanager - Passive服务

export DISPLAY=192.168.0.11:0.0     # 宿主机ip
xhost +

cd $ORACLE_HOME
./runInstaller

4.4 配置监听

export DISPLAY=192.168.0.11:0.0
xhost +
netca

4.5 建库

export DISPLAY=192.168.0.11:0.0
xhost +
dbca

5 其他调整

5.1 sqlplus配置

cat>>$ORACLE_HOME/sqlplus/admin/glogin.sql<<EOF
set linesize 120
set pagesize 999
define _editor=vi
set sqlprompt "_user'@'_connect_identifier> "
EOF

5.2 rlwrap安装配置

yum -y install readline*

# 上传rlwrap包
tar -xvf rlwrap-0.37.tar.gz
cd rlwrap-0.37
./configure
make && make install
which rlwrap

cat>>/home/oracle/.bash_profile<<EOF
alias sqlplus='/usr/local/bin/rlwrap sqlplus'
alias rman='/usr/local/bin/rlwrap rman'
EOF
 
# oracle用户
source ~/.bash_profile
